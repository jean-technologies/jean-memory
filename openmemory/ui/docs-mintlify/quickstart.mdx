This is the fastest way to add a complete, context-aware AI chatbot to your React application.

```jsx {{ title: 'pages/index.tsx' }}
import { JeanProvider, JeanChat, SignInWithJean } from 'jeanmemory-react';

export default function MyPage() {
  return (
    <JeanProvider apiKey="YOUR_API_KEY">
      <JeanChat />
    </JeanProvider>
  );
}
```

<br/>
<Tip>
The `<JeanChat />` component includes the `<SignInWithJean />` button and handles the entire authentication and chat experience out of the box. For more advanced use cases, including headless backend integration, check out our full [SDK documentation](/sdk/overview).
</Tip>

<script
  dangerouslySetInnerHTML={{
    __html: `
      const copyToClipboard = (text) => {
        if (navigator.clipboard) {
          navigator.clipboard.writeText(text).then(() => {
            console.log('Docs copied to clipboard');
          }, (err) => {
            console.error('Failed to copy: ', err);
          });
        }
      };

      const btn = document.getElementById('copyDocsBtn');
      if (btn) {
        btn.onclick = () => {
          fetch('/assets/consolidated-docs.md')
            .then(response => response.text())
            .then(data => {
              copyToClipboard(data);
              btn.innerHTML = 'Copied!';
              setTimeout(() => {
                btn.innerHTML = 'Copy All Docs';
              }, 2000);
            });
        };
      }
    `,
  }}
/>
