Choose your path. Add a complete UI component to your frontend or add a powerful context layer to your backend.

<CodeGroup>
```jsx {{ title: 'UI Appliance (React)' }}
// 1. Install the React SDK
// npm install jeanmemory-react

// 2. Add the provider and chat component
import { JeanProvider, JeanChat } from 'jeanmemory-react';

function MyPage() {
  return (
    <JeanProvider apiKey="YOUR_API_KEY">
      <JeanChat />
    </JeanProvider>
  );
}
```
```python {{ title: 'Headless Backend (Python)' }}
# 1. Install the Python SDK
# pip install jeanmemory openai

# 2. Get context before calling your LLM
from jeanmemory import JeanClient
from openai import OpenAI

jean = JeanClient(api_key="YOUR_JEAN_API_KEY")
openai = OpenAI(api_key="YOUR_OPENAI_API_KEY")

context = jean.get_context(
    user_token="USER_TOKEN_FROM_FRONTEND",
    message="What was our last conversation about?"
).text

# 3. Use context in your LLM call
completion = openai.chat.completions.create(
    model="gpt-4-turbo",
    messages=[{"role": "user", "content": f"Context: {context}\\n\\nUser question: What was our last conversation about?"}]
)
```

<script

    __html: `
      const copyToClipboard = (text) => {
        if (navigator.clipboard) {
          navigator.clipboard.writeText(text).then(() => {
            console.log('Docs copied to clipboard');
          }, (err) => {
            console.error('Failed to copy: ', err);
          });
        }
      };

      const btn = document.getElementById('copyDocsBtn');
      if (btn) {
        btn.onclick = () => {
          fetch('/assets/consolidated-docs.md')
            .then(response => response.text())
            .then(data => {
              copyToClipboard(data);
              btn.innerHTML = 'Copied!';
              setTimeout(() => {
                btn.innerHTML = 'Copy All Docs';
              }, 2000);
            });
        };
      }
    `,
  }}
